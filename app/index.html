אָרִיאַדְנֶה: הַי לְכֻלָּם בַּצֶּ'ט הַזֶּה! אֲנַחְנוּ מְחַפְּשִׂים מִתְנַדְּבִים אַמִּיצִים כְּדֵי לְבַצֵּעַ אֶת הַמְּשִׂימָה הַבָּאָה – לַעֲבֹר דֶּרֶךְ מָבוֹךְ, וְהָאָדָם שֶׁיַּצְלִיחַ יְקַבֵּל פְּרָס מְיֻחָד בְּיוֹתֵר – הוּא יִתְחַתֵּן אִתִּי! תֵזֵאוּס: תַּקְשִׁיבוּ לִי – אַל תֵּלְכוּ לְשָׁם! הַהַרְפַּתְקָה הַזֹּאת לֹא תִּגָּמֵר טוֹב – לְכָל הַפָּחוֹת זֶה יִגָּמֵר בְּנִתּוּק מֻחְלָט מֵהַמִּשְׁפָּחָה, אוֹ יוֹתֵר גָּרוּעַ. אָרִיאַדְנֶה: תֵזֵאוּס! אַל תִּהְיֶה חָצוּף! מִי שֶׁצָּרִיךְ לְהִתְנַתֵּק זֶה אַתָּה, מֵהַסִּפּוּר הַזֶּה וּמֵהַצֶּ'ט הַזֶּה. לִפְנֵי שֶׁבִּקַּרְתָּ אֶצְלֵנוּ וְהִתְחַלְתָּ לְהָפִיץ שְׁמוּעוֹת לֹא מֻצְדָּקוֹת מִסְפַּר הַמְּבַקְּרִים אֶצְלֵנוּ שָׁבַר שִׂיאִים, וְעַכְשָׁו מִסְפַּר הַמֻּעֲמָדִים יָרַד בִּגְלָלְךָ חֲבָל עַל הַזְּמַן1. תֵזֵאוּס: אֵיזוֹ מִסְכֵּנָה! תֵּרָגְעִי! אֲנִי יָכוֹל לְהַסְבִּיר אֶת הַנְּטִיָּה הַמַּפְתִּיעָה הַזֹּאת בְּקַלּוּת. זֶה בִּגְלַל שֶׁשּׁוּם דָּבָר טוֹב לֹא קוֹרֶה שָׁם. זֹאת הַסִּבָּה בִּגְלָלָהּ מִסְפַּר הַמְּבַקְּרִים שֶׁלָּכֶם יוֹרֵד. מֻעֲמָד 1: אַתֶּם בְּסִין?

אָרִיאַדְנֶה: לֹא! אֲנַחְנוּ בְּיָוָן. מַמָּשׁ לְיַד הַיָּם הַתִּיכוֹן. תֵזֵאוּס: אַתְּ קוֹרֵאת לַשְּׁלוּלִית הַזֹּאת יָם? מֻעֲמָד 1: יֹפִי! אִם הֲיִיתֶם גָּרִים בְּסִין הָיִיתִי צָרִיךְ לְהָבִיא מְתַרְגֵּם. :) תֵזֵאוּס: תִּבְדֹּק אֶת הַכֹּל טוֹב טוֹב לִפְנֵי שֶׁאַתָּה נוֹסֵעַ! מֻעֲמָד 1: אָרִיאַדְנֶה, יֵשׁ בְּיָוָן תַּחְבּוּרָה צִבּוּרִית? יֵשׁ הַרְבֵּה פְּקָקִים בַּדְּרָכִים? אָרִיאַדְנֶה: אֲנַחְנוּ יְכוֹלִים לְהַסִּיעַ אוֹתְךָ מִנְּמַל הַתְּעוּפָה! תֵזֵאוּס: מֻעֲמָד 1, אַל תֵּלֵךְ! אֲנִי אוֹמֵר לְךָ, זֶה מַמָּשׁ גְּזַר דִּין מָוֶת2! מֻעֲמָד 1: תֵזֵאוּס, לָמָּה כָּל הַלַּחַץ? תִּשְׁמַע, אֲנִי אוֹהֵב לִזְרֹם3. אֲנִי לֹא מֵבִין, מָה הַבְּעָיָה שֶׁלְּךָ? כָּל הַסִּפּוּר, בְּקִצּוּר – בַּחוּרָה יָפָה וּמֻכְשֶׁרֶת מַזְמִינָה אוֹתִי לְהִתְאָרֵחַ אֶצְלָהּ, זֶה הַכֹּל. אוּלַי אֶצְלְךָ זֶה קוֹרֶה כָּל שֵׁנִי וַחֲמִישִׁי4, אֲבָל בָּעוֹלָם שֶׁלִּי זֶה מַשֶּׁהוּ יוֹצֵא מִן הַכְּלָל. זֶה סוּג שֶׁל5 נֵס. מֻעֲמָד 1: אָרִיאַדְנֶה, אֲנִי בָּא תּוֹךְ יוֹמַיִם. תִּסְמְכִי עָלַי, אֲנִי בָּא בַּזְּמַן. תֵזֵאוּס: מֻעֲמָד 1, שׁוּם דָּבָר טוֹב לֹא יֵצֵא6 מִזֶּה. אִם אַתָּה יֶלֶד יָחִיד אֲנִי מְרַחֵם עַל הַהוֹרִים שֶׁלְּךָ. מֻעֲמָד 1: תֵזֵאוּס! תַּשְׁקִיעַ אֶת הַזְּמַן שֶׁלְּךָ בְּמַשֶּׁהוּ אַחֵר. אָרִיאַדְנֶה: מֻעֲמָד 1, אֵיךְ קוֹרְאִים לְךָ? מֻעֲמָד 1: מִינוֹטָאוּר.
